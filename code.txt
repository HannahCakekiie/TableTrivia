        function loadQuestion() {
            if (currentQuestion < bigBangQuiz.length) {
                document.getElementById("question").textContent = bigBangQuiz[currentQuestion].question;
                document.getElementById("trivia").textContent = "";
                const choicesContainer = document.getElementById("choices");
                choicesContainer.innerHTML = "";
                bigBangQuiz[currentQuestion].choices.forEach(choice => {
                    const button = document.createElement("button");
                    button.textContent = choice;
                    button.onclick = () => checkAnswer(button, choice);
                    choicesContainer.appendChild(button);
                });
            } else {
                showResults();
            }
        }
        function checkAnswer(button, choice) {
            document.getElementById("trivia").textContent = bigBangQuiz[currentQuestion].trivia;
            if (choice === bigBangQuiz[currentQuestion].answer) {
                button.classList.add("correct");
                score++;
            } else {
                button.classList.add("wrong");
                const correctButton = [...document.getElementById("choices").children].find(btn => btn.textContent === bigBangQuiz[currentQuestion].answer);
                if (correctButton) correctButton.classList.add("correct-answer");
            }
            currentQuestion++;
            setTimeout(loadQuestion, 5000);
        }
        function showResults() {
            let scoreCategory = "";
            if (score <= 5) {
                scoreCategory = "Needs Improvement";
            } else if (score <= 10) {
                scoreCategory = "Good";
            } else {
                scoreCategory = "Excellent";
            }
            document.getElementById("score-category").textContent = scoreCategory;
            document.getElementById("quiz-box").classList.remove("visible");
            document.getElementById("end-screen").classList.add("visible");
            document.getElementById("score").textContent = score;
        }

        function startQuiz() {
            console.log("Quiz is starting");

            if (!checkCoins()) {
                return; // Exit if not enough coins
            }

            document.getElementById("start-screen").style.display = "none";
            document.getElementById("quiz-box").style.display = "block";
            loadQuestion();
        }

        function loadQuestion() {
            if (currentQuestion < periodicQuiz.length) {
                const q = periodicQuiz[currentQuestion];
                document.getElementById("question").textContent = q.question;
                document.getElementById("trivia").textContent = "";
                const choicesContainer = document.getElementById("choices");
                choicesContainer.innerHTML = "";
                q.choices.forEach(choice => {
                    const button = document.createElement("button");
                    button.textContent = choice;
                    button.onclick = () => checkAnswer(button, choice);
                    choicesContainer.appendChild(button);
                });
            } else {
                showResults();
            }
        }

        function checkAnswer(button, choice) {
            const q = periodicQuiz[currentQuestion];
            document.getElementById("trivia").textContent = q.trivia;
            if (choice === q.answer) {
                button.classList.add("correct");
                score++;
            } else {
                button.classList.add("wrong");
                const correctButton = [...document.getElementById("choices").children].find(btn => btn.textContent === q.answer);
                if (correctButton) correctButton.classList.add("correct-answer");
            }
            userCoins = 0; // Reset coins to zero
            currentQuestion++;

            setTimeout(loadQuestion, 2000);
        }

        function showResults() {
            document.getElementById("quiz-box").style.display = "none";
            document.getElementById("end-screen").style.display = "block";
            document.getElementById("score").textContent = score + " / 20";
            document.getElementById("total-questions").textContent = "20";

            // Determine score category
            let scoreCategory;
            if (score <= 3) {
                scoreCategory = "Very Bad";
            } else if (score <= 7) {
                scoreCategory = "Bad";
            } else if (score <= 11) {
                scoreCategory = "Good";
            } else {
                scoreCategory = "Very Good";
            }
            document.getElementById("score-category").textContent = "Score Category: " + scoreCategory;
        }
